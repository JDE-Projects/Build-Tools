# Vetted per-app extras menu for JDE-Projects Simple X Tools.
# This file is NOT installed wholesale. When a new tool needs one of these,
# copy the exact lines it imports into that tool's own requirements.txt.
# Dependabot (see ../../.github/dependabot.yml) bumps these pins here so the
# menu a new tool copies from stays current, and so the shipped crypto stays
# patched. Do not hand-pin these from a local pip install.
# Proven against the fleet's green CI builds (Python 3.14).

PyNaCl==1.6.2

# Optional "remember password" via the OS keychain (Windows Credential Manager).
keyring==25.7.0
